<script>
  const API = 'https://mintkit-production.up.railway.app';
  const COLORS = ['#185fa5','#3b6d11','#854f0b','#7C3AED','#DC2626','#0891b2'];

  async function load() {
    try {
      const [statusRes, coinsRes] = await Promise.all([
        fetch(`${API}/api/status`),
        fetch(`${API}/api/coins`)
      ]);
      const status = await statusRes.json();
      const { coins } = await coinsRes.json();

      document.getElementById('stat-coins').textContent    = status.tokens_deployed ?? 0;
      document.getElementById('stat-burned').textContent   = (status.tokens_burned ?? 0).toLocaleString();
      document.getElementById('stat-buybacks').textContent = status.buybacks_executed ?? 0;

      if (coins && coins.length > 0) {
        const grid = document.getElementById('coins-grid');
        grid.innerHTML = '';
        coins.forEach((coin, i) => {
          const color   = COLORS[i % COLORS.length];
          const net     = coin.network === 'mainnet' ? '' : '?cluster=devnet';
          const explorer = `https://explorer.solana.com/address/${coin.mint_address}${net}`;
          grid.innerHTML += `
            <div class="coin-card">
              <div class="coin-head">
                <div class="coin-icon" style="background:${color};">${coin.ticker[0]}</div>
                <div>
                  <div class="coin-title">${coin.coin_name}</div>
                  <div class="coin-sub">$${coin.ticker}</div>
                </div>
              </div>
              <div class="coin-tagline">Launched transparently using MintKit — all tokenomics on-chain and verifiable.</div>
              <div class="coin-grid">
                <div class="coin-cell"><div class="cell-label">Supply</div><div class="cell-value">${(coin.total_supply||0).toLocaleString()}</div></div>
                <div class="coin-cell"><div class="cell-label">Network</div><div class="cell-value">${coin.network}</div></div>
                <div class="coin-cell"><div class="cell-label">LP lock</div><div class="cell-value">180 days</div></div>
                <div class="coin-cell"><div class="cell-label">Launched</div><div class="cell-value">${(coin.deployed_at||'').slice(0,10)}</div></div>
              </div>
              <div class="badges">
                <span class="badge b-green">LP locked</span>
                <span class="badge b-blue">Mint revoked</span>
                <span class="badge b-amber">Auto buyback</span>
              </div>
              <div class="links">
                <a href="${explorer}" class="link-btn" target="_blank">Explorer</a>
                <a href="https://github.com/TRNDMNTBT/MintKit" class="link-btn" target="_blank">Transparency log</a>
              </div>
            </div>`;
        });
      }
    } catch(e) {
      document.getElementById('stat-coins').textContent    = '0';
      document.getElementById('stat-burned').textContent   = '0';
      document.getElementById('stat-buybacks').textContent = '0';
    }
  }

  load();
</script>
